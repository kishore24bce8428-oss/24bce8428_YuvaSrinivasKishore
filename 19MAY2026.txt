USE VITBranches;
SHOW TABLES FROM VITBranches;
SELECT *
FROM Worker;

SELECT FIRST_NAME,DEPARTMENT,SALARY
FROM Worker
WHERE DEPARTMENT='HR' AND SALARY>100000;

SELECT FIRST_NAME,DEPARTMENT,SALARY
FROM Worker
WHERE DEPARTMENT='HR' OR SALARY>100000;

SELECT WORKER_ID,FIRST_NAME,DEPARTMENT,SALARY
FROM Worker
WHERE WORKER_ID IN (1,4,3);

SELECT WORKER_ID,FIRST_NAME,DEPARTMENT,SALARY
FROM Worker
WHERE WORKER_ID NOT IN (2,5,7,6,8);

SELECT *
FROM Worker
WHERE SALARY BETWEEN 100000 AND 500000;

SELECT *
FROM Worker
WHERE SALARY NOT BETWEEN 50000 AND 90000;

SELECT *
FROM Worker
WHERE SALARY BETWEEN 100000 AND 500000
			AND DEPARTMENT NOT IN ('HR','Account');

SELECT *
FROM Worker
WHERE FIRST_NAME LIKE 'A%';

SELECT *
FROM Worker
WHERE FIRST_NAME LIKE '%L';

SELECT *
FROM Worker
WHERE FIRST_NAME LIKE '_i%';

SELECT *
FROM Worker
WHERE FIRST_NAME LIKE '%sh%';

SELECT *
FROM Worker
WHERE DEPARTMENT='Admin'
		AND SALARY IN (SELECT MIN(SALARY) FROM Worker WHERE DEPARTMENT='Admin' AND FIRST_NAME NOT LIKE 'A%')
		AND FIRST_NAME NOT LIKE 'A%';

SELECT MIN(SALARY)
FROM Worker
WHERE DEPARTMENT='HR';

SELECT MAX(SALARY)
FROM Worker
WHERE DEPARTMENT='HR';

SELECT SUM(SALARY)
FROM Worker
WHERE DEPARTMENT='HR';

SELECT COUNT(WORKER_ID)
FROM Worker
WHERE DEPARTMENT='Account';

SELECT AVG(SALARY)
FROM Worker
WHERE DEPARTMENT='Account';

SELECT SALARY FROM Worker
UNION
SELECT SALARY FROM Worker;

SELECT SALARY FROM Worker
UNION ALL
SELECT SALARY FROM Worker;

SELECT WORKER_ID AS ROLL_NUMBER
FROM Worker;

SELECT DISTINCT(DEPARTMENT)
FROM Worker;

SELECT WORKER_ID,FIRST_NAME,DEPARTMENT,
CASE
	WHEN SALARY>300000
			THEN 'High Salary'
	WHEN SALARY BETWEEN 100000 AND 300000
			THEN 'Medium Salary'
	ELSE 'Low Salary'
END
AS SALARY_STATUS
FROM Worker;

SELECT * FROM Worker;

CREATE VIEW MinSalariedWorkers AS
SELECT FIRST_NAME,LAST_NAME,SALARY,DEPARTMENT
FROM Worker
WHERE SALARY IN (SELECT MIN(SALARY) FROM Worker GROUP BY DEPARTMENT); 

CREATE VIEW MaxSalariedWorkers AS
SELECT FIRST_NAME,LAST_NAME,SALARY,DEPARTMENT
FROM Worker
WHERE SALARY IN (SELECT MAX(SALARY) FROM Worker GROUP BY DEPARTMENT); 

SELECT *
FROM MinSalariedWorkers;

SELECT *
FROM MaxSalariedWorkers;
